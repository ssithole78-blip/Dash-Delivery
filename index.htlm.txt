<script>
function sendToWhatsApp(event) {
    event.preventDefault();

    // Phone number without the '+' sign for clean URL formatting
    const phone = "263772447516"; 
    const service = document.getElementById('service').value;
    const pickup = document.getElementById('pickup').value;
    const dropoff = document.getElementById('dropoff').value;
    const details = document.getElementById('details').value || "N/A";
    const name = document.getElementById('name').value;

    // Clean JavaScript template string syntax
    const rawMessage = *NEW DASH DELIVERY ORDER* ⚡\n\n +
                       👤 *Customer:* ${name}\n +
                       📦 *Service:* ${service}\n +
                       📍 *Pickup:* ${pickup}\n +
                       🏁 *Drop-off:* ${dropoff}\n +
                       📝 *Details:* ${details};

    // Safely encode the entire message
    const encodedMessage = encodeURIComponent(rawMessage);

    // Use clean wa.me link
    const whatsappUrl = https://wa.me/${phone}?text=${encodedMessage};
    
    // Redirect user to WhatsApp
    window.location.href = whatsappUrl;
}
